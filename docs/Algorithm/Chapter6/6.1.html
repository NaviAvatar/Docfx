    6.1 图的一些算法

    如果判断一个五向图是不是有环？

    答：

    ① 将所有度小于等于1的顶点入队。

    ②当队不为空时，在队中取一个元素A（访问），与A的相连接的结点度减1（消去各顶点与A的连接）。

    ③将度为1且不在队中的顶点入队。度不为空，往①循环。

    当循环结束时，访问顶点数等于全部顶点个数时，说明无环，否则有环。

    #pragma once
#include <map>
#include <vector>
#include<queue>
#include<set>

using namespace std;

// 无向图类,采用邻接表法，这里暂时不考虑性能问题
class MapLearnning {
private:
	map<int, vector<int>> data;

public:
	//	将边加入 
	void AddEdge(int s, int e)
	{
		data[s].push_back(e);
		data[e].push_back(s);
	}

	// 判断是否有环
	bool HaveARing()
	{
		map<int, vector<int>> tmp = data;	// 拷贝一份数据
		int count = tmp.size();				// 顶点个数
		vector<int>	que;					// 辅助队列，队列不能遍历好傻逼
		set<int> hasVisit;				// 访问过结点的
		for (auto& x : tmp)
		{
			if (x.second.size() <= 1)	// 将所有度小于等于1的顶点入队
				que.push_back(x.first);
		}

		while (!que.empty()) {
			auto element = que.back();
			que.erase(--(que.end()));

			hasVisit.insert(element);
			count--;
			// 与 element 相连的结点消去边的影响
			for (auto& v : tmp)
			{
				auto pBeg = v.second.begin();
				auto pEnd = v.second.end();

				while (pBeg != pEnd)
				{
					if (*pBeg == element)
						break;
					pBeg++;
				}

				// 消边
				if (pBeg != pEnd)
					v.second.erase(pBeg);

				// 当前结点v.second未访问过
				if (hasVisit.find(v.first) == hasVisit.end() && v.second.size() <= 1)
				{
					if(std::find(std::begin(que), std::end(que), v.first)==std::end(que))
						que.push_back(v.first);
				}
					
			}

			// 清空队头结点与所有结点的连接
			tmp[element].clear();
		 }

		return !(count == 0);
	}
};


TestCase()
{
    MapLearnning mapl;

    mapl.AddEdge(1, 2);
    mapl.AddEdge(1, 5);
    mapl.AddEdge(2, 6);
    // mapl.AddEdge(3, 4);
    mapl.AddEdge(3, 6);
   // mapl.AddEdge(3, 7);
    //mapl.AddEdge(4, 7);
    mapl.AddEdge(4, 8);
    mapl.AddEdge(6, 7);
    mapl.AddEdge(7, 8);

    if (mapl.HaveARing())
        cout << " 有环" << endl;
    else
        cout << " 无环" << endl;
}
